RPerl Install Notes
Single-User Installation (Not System-Wide Installation)

$ mkdir -p ~/RPerl/3rd_party/CPAN
$ cd ~/.cpan/CPAN
$ mv MyConfig.pm MyConfig.pm.old
$ ln -s path/to/RPerl/lib/CPAN/MyConfig.pm MyConfig.pm
$ ln -s path/to/RPerl /tmp/RPerl-latest

$ cpan
cpan> o conf
	confirm RPerl entries are loaded
cpan> install Parse::RecDescent
cpan> force install Inline
cpan> force install Inline::CPP
	must force, Inline tests are broken, they change @INC and then fail to load single-user-installation of Parse::RecDescent
cpan> force install Inline::Filters

$ mv ~/RPerl/3rd_party/CPAN/usr/local/share/perl/5.10.1/Inline/Filters.pm ~/RPerl/3rd_party/CPAN/usr/local/share/perl/5.10.1/Inline/Filters.pm.old
$ ln -s path/to/RPerl/lib/Inline/Filters.pm ~/RPerl/3rd_party/CPAN/usr/local/share/perl/5.10.1/Inline/Filters.pm

$ mv /usr/share/perl/5.10.1/ExtUtils/typemap /usr/share/perl/5.10.1/ExtUtils/typemap.old
$ ln -s path/to/RPerl/lib/ExtUtils/typemap /usr/share/perl/5.10.1/ExtUtils/typemap
